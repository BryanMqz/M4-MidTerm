49 """

52 Procesa la imagen capturada, detecta el objeto verde y calcula sus coordenadas.

58 cv_image = self.bridge.imgmsg_to_cv2(data, "bgr8")

60 mask = cv2.inRange(hsv_image, self.green_lower, self.green_upper)
61 contours, _ = cv2.findContours(mask, cv2.RETR_EXTERNAL, cv2.CHAIN_APPROX_SIMPLE)
62 if contours:
63 largest_contour = max(contours, key=cv2.contourArea)
